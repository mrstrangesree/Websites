from pyspark.sql.functions import (col, when, lit, cast)

# Add / overwrite column
df.withColumn("tax", col("price") * 0.1)

df.withColumn("grade",
    when(col("score") >= 90, lit("A"))
   .when(col("score") >= 80, lit("B"))
   .when(col("score") >= 70, lit("C"))
   .otherwise(lit("F")))

# Multiple withColumns (Spark 3.3+)
df.withColumns({
    "tax":     col("price") * 0.1,
    "revenue": col("price") * col("qty")})

# IIF (inline if)
from pyspark.sql.functions import iif   # Spark 3.4+
df.withColumn("is_adult", iif(col("age") >= 18, lit(True), lit(False)))

📊Aggregations — GROUP BY / HAVING
PySpark
from pyspark.sql.functions import (
    count, countDistinct, sum, avg, min, max,
    collect_list, collect_set, first, last,
    stddev, variance, skewness, kurtosis,
    approx_count_distinct, percentile_approx)

# Basic groupBy + agg
df.groupBy("dept").agg(
    count("*").alias("cnt"),
    countDistinct("emp_id").alias("uniq_emp"),
    sum("salary").alias("total_sal"),
    avg("salary").alias("avg_sal"),
    min("salary").alias("min_sal"),
    max("salary").alias("max_sal"))

# HAVING equivalent — filter after agg
(df.groupBy("dept")
   .agg(avg("salary").alias("avg_sal"))
   .filter(col("avg_sal") > 50000))

# Collect values into array
df.groupBy("dept").agg(
    collect_list("name").alias("names"),   # with dupes
    collect_set("name").alias("uniq_names"))# no dupes

# Pivot
df.groupBy("dept").pivot("year",
    [2022,2023,2024]).sum("sales")

# ROLLUP  (subtotals)
df.rollup("dept","city").agg(sum("sales"))

# CUBE    (all combos)
df.cube("dept","city").agg(sum("sales"))

# Approximate count (faster on large data)
df.groupBy("country").agg(
    approx_count_distinct("user_id", 0.05))
Spark SQL
-- Basic GROUP BY
SELECT dept,
  COUNT(*)               AS cnt,
  COUNT(DISTINCT emp_id) AS uniq_emp,
  SUM(salary)            AS total_sal,
  AVG(salary)            AS avg_sal,
  MIN(salary)            AS min_sal,
  MAX(salary)            AS max_sal
FROM employees
GROUP BY dept;

-- HAVING
SELECT dept, AVG(salary) AS avg_sal
FROM employees
GROUP BY dept
HAVING AVG(salary) > 50000;

-- Collect values
SELECT dept,
  COLLECT_LIST(name) AS names,
  COLLECT_SET(name)  AS uniq_names
FROM employees GROUP BY dept;

-- PIVOT
SELECT * FROM (SELECT dept, year, sales FROM t)
PIVOT (SUM(sales) FOR year IN (2022,2023,2024));

-- ROLLUP
SELECT dept, city, SUM(sales)
FROM t GROUP BY ROLLUP(dept, city);

-- CUBE
SELECT dept, city, SUM(sales)
FROM t GROUP BY CUBE(dept, city);

-- GROUPING SETS (manual control)
SELECT dept, city, SUM(sales)
FROM t GROUP BY GROUPING SETS(
  (dept,city),(dept),(city),());

🪟Window Functions
PySpark
from pyspark.sql.functions import (
    row_number, rank, dense_rank, percent_rank,
    ntile, lag, lead, first, last,
    sum, avg, min, max, count,
    unboundedPreceding as UNBD_PRE,
    unboundedFollowing as UNBD_FOL,
    currentRow as CUR)
from pyspark.sql.window import Window

# Define window spec
w_dept = (Window
    .partitionBy("dept")        # group
    .orderBy(col("salary").desc()))

# Ranking functions
df.withColumn("rn",     row_number().over(w_dept))
df.withColumn("rank",   rank().over(w_dept))      # gaps
df.withColumn("drank",  dense_rank().over(w_dept))# no gaps
df.withColumn("pct_r",  percent_rank().over(w_dept))
df.withColumn("tile",   ntile(4).over(w_dept))    # quartile

# LAG / LEAD (offset rows)
w_time = Window.partitionBy("id").orderBy("date")
df.withColumn("prev_val", lag("value", 1).over(w_time))
df.withColumn("next_val", lead("value", 1).over(w_time))

# Running totals / cumulative sum
w_cum = (Window.partitionBy("dept").orderBy("date")
         .rowsBetween(UNBD_PRE, CUR))
df.withColumn("running_sum", sum("sales").over(w_cum))

# Rolling 7-day average
w_roll = (Window.partitionBy("id").orderBy("date")
          .rowsBetween(-6, CUR))
df.withColumn("roll7_avg", avg("value").over(w_roll))

# Full partition aggregate (no ORDER BY)
w_part = Window.partitionBy("dept")
df.withColumn("dept_total", sum("sales").over(w_part))
df.withColumn("dept_avg",   avg("salary").over(w_part))

# Top-N per group
top2 = (df.withColumn("rn", row_number().over(w_dept))
          .filter(col("rn") <= 2))
Spark SQL
-- Ranking
SELECT *, 
  ROW_NUMBER() OVER(PARTITION BY dept ORDER BY salary DESC) AS rn,
  RANK()       OVER(PARTITION BY dept ORDER BY salary DESC) AS rnk,
  DENSE_RANK() OVER(PARTITION BY dept ORDER BY salary DESC) AS drnk,
  PERCENT_RANK()OVER(PARTITION BY dept ORDER BY salary DESC)AS pct,
  NTILE(4)     OVER(PARTITION BY dept ORDER BY salary DESC) AS quartile
FROM employees;

-- LAG / LEAD
SELECT *,
  LAG(value,  1) OVER(PARTITION BY id ORDER BY date) AS prev_val,
  LEAD(value, 1) OVER(PARTITION BY id ORDER BY date) AS next_val
FROM series;

-- Running total (cumulative sum)
SELECT *,
  SUM(sales) OVER(
    PARTITION BY dept ORDER BY date
    ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
  ) AS running_sum
FROM sales;

-- 7-day rolling average
SELECT *,
  AVG(value) OVER(
    PARTITION BY id ORDER BY date
    ROWS BETWEEN 6 PRECEDING AND CURRENT ROW
  ) AS roll7_avg
FROM metrics;

-- Partition total (full window, no order)
SELECT *,
  SUM(sales) OVER(PARTITION BY dept) AS dept_total,
  AVG(salary)OVER(PARTITION BY dept) AS dept_avg
FROM employees;

-- Top-N per group (CTE approach)
WITH ranked AS (
  SELECT *, ROW_NUMBER() OVER(
    PARTITION BY dept ORDER BY salary DESC) AS rn
  FROM employees)
SELECT * FROM ranked WHERE rn <= 2;

# ── Skew handling — Salting technique ─────────────────
import pyspark.sql.functions as F
from pyspark.sql.functions import lit, rand, concat, floor

SALT = 20   # number of salt buckets

# Salt the skewed key (large side)
skewed = df_big.withColumn(
    "salt_key",
    concat(col("key"), lit("_"),
           (rand()*SALT).cast("int").cast("string")))

# Explode the small side to match all salts
small_exp = df_small.withColumn(
    "salt_n", F.explode(sequence(lit(0), lit(SALT-1))))
small_exp = small_exp.withColumn(
    "salt_key",
    concat(col("key"), lit("_"), col("salt_n").cast("string")))

skewed.join(small_exp, "salt_key")

# AQE skew join (automatic — no code needed)
spark.conf.set("spark.sql.adaptive.enabled","true")
spark.conf.set("spark.sql.adaptive.skewJoin.enabled","true")
