✦ Delta Lake – MERGE (Upsert)
# UPSERT with MERGE (SCD Type 1 pattern)
from delta.tables import DeltaTable

target = DeltaTable.forPath(spark, "/mnt/delta/sales")

target.alias("t").merge(
    df.alias("s"),
    "t.order_id = s.order_id"   # join key
).whenMatchedUpdateAll()     # update if match\
 .whenNotMatchedInsertAll()  # insert if new\
 .execute()

# Conditional update on match
target.alias("t").merge(df.alias("s"), "t.id = s.id")\
  .whenMatchedUpdate(
    condition = "s.updated_at > t.updated_at",
    set = {"amount": "s.amount"}
  ).whenNotMatchedInsertAll().execute()

✦ Structured Streaming
# Read stream from Kafka topic
stream_df = (spark.readStream
  .format("kafka")
  .option("kafka.bootstrap.servers", "broker:9092")
  .option("subscribe", "orders-topic")
  .option("startingOffsets", "latest")
  .load())

# Parse JSON payload from Kafka value bytes
from pyspark.sql.functions import from_json, col
from pyspark.sql.types import StructType, StringType, DoubleType
schema = StructType().add("order_id", StringType()).add("amount", DoubleType())

parsed = stream_df\
  .selectExpr("CAST(value AS STRING) as json_str")\
  .select(from_json(col("json_str"), schema).alias("data"))\
  .select("data.*")

# 5-minute tumbling window aggregation
from pyspark.sql.functions import window
agg = parsed.groupBy(window("event_time", "5 minutes"), "region")\
             .sum("amount")

# Write stream to Delta table
query = (parsed.writeStream
  .format("delta")
  .outputMode("append")
  .option("checkpointLocation", "/mnt/checkpoints/orders")
  .toTable("analytics.orders_stream"))

# Write stream to console (debug only)
query = parsed.writeStream.format("console").start()

# Stop stream gracefully
query.stop()         # or query.awaitTermination() to block

🪟 Window Functions · 🤖 MLflow
✦ Window Functions
# Define a window: partition by region, order by date
from pyspark.sql.window import Window
from pyspark.sql.functions import (
    row_number, rank, dense_rank,
    lag, lead, sum as wsum
)

w = Window.partitionBy("region")\
           .orderBy("order_date")

# Row number (unique sequential rank)
df = df.withColumn("row_num", row_number().over(w))

# Rank (ties share same rank, next skipped)
df = df.withColumn("rnk", rank().over(w))

# Dense rank (ties share same rank, no skip)
df = df.withColumn("dense_rnk", dense_rank().over(w))

# Running total (cumulative sum)
w_cum = Window.partitionBy("region")\
              .orderBy("order_date")\
              .rowsBetween(Window.unboundedPreceding,
                           Window.currentRow)
df = df.withColumn("running_total",
    wsum("amount").over(w_cum))

# Lag (previous row value) / Lead (next row value)
df = df.withColumn("prev_amt", lag("amount", 1).over(w))
df = df.withColumn("next_amt", lead("amount", 1).over(w))
